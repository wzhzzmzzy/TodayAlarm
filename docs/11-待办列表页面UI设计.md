// ui/screens/TodoListScreen.kt
@Composable
fun TodoListScreen(
    uiState: TodoListUiState,
    onAction: (TodoListAction) -> Unit,
    modifier: Modifier = Modifier
) {
    val windowInfo = rememberTodoWindowInfo()

    Box(modifier = modifier.fillMaxSize()) {
        when (windowInfo.screenWidthInfo) {
            TodoWindowInfo.WindowType.Compact -> {
                CompactTodoListLayout(uiState, onAction)
            }
            TodoWindowInfo.WindowType.Medium -> {
                MediumTodoListLayout(uiState, onAction)
            }
            TodoWindowInfo.WindowType.Expanded -> {
                ExpandedTodoListLayout(uiState, onAction)
            }
        }
    }
}

@Composable
private fun CompactTodoListLayout(
    uiState: TodoListUiState,
    onAction: (TodoListAction) -> Unit
) {
    Column(modifier = Modifier.fillMaxSize()) {
        // 顶部应用栏
        TodoListTopBar(
            searchQuery = uiState.searchQuery,
            onSearchQueryChange = { onAction(TodoListAction.SearchTodoItems(it)) },
            onFilterClick = { onAction(TodoListAction.ShowFilterDialog) },
            onSortClick = { onAction(TodoListAction.ShowSortDialog) }
        )

        // 筛选条件栏
        TodoFilterChips(
            currentFilter = uiState.currentFilter,
            onFilterChange = { onAction(TodoListAction.UpdateFilter(it)) }
        )

        // 待办列表
        TodoListContent(
            uiState = uiState,
            onAction = onAction,
            modifier = Modifier.weight(1f)
        )
    }

    // 浮动操作按钮
    Box(
        modifier = Modifier.fillMaxSize(),
        contentAlignment = Alignment.BottomEnd
    ) {
        FloatingActionButton(
            onClick = { onAction(TodoListAction.NavigateToCreate) },
            modifier = Modifier.padding(TodoDimensions.SpacingMedium)
        ) {
            Icon(
                imageVector = Icons.Default.Add,
                contentDescription = "添加待办"
            )
        }
    }
}

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun TodoListTopBar(
    searchQuery: String?,
    onSearchQueryChange: (String) -> Unit,
    onFilterClick: () -> Unit,
    onSortClick: () -> Unit,
    modifier: Modifier = Modifier
) {
    var isSearchActive by remember { mutableStateOf(false) }

    TopAppBar(
        title = {
            if (isSearchActive) {
                SearchTextField(
                    query = searchQuery ?: "",
                    onQueryChange = onSearchQueryChange,
                    onSearchActiveChange = { isSearchActive = it },
                    placeholder = "搜索待办事项..."
                )
            } else {
                Text(
                    text = "待办事项",
                    style = MaterialTheme.typography.headlineSmall
                )
            }
        },
        actions = {
            if (!isSearchActive) {
                IconButton(onClick = { isSearchActive = true }) {
                    Icon(
                        imageVector = Icons.Default.Search,
                        contentDescription = "搜索"
                    )
                }
            }

            IconButton(onClick = onFilterClick) {
                Icon(
                    imageVector = Icons.Default.FilterList,
                    contentDescription = "筛选"
                )
            }

            IconButton(onClick = onSortClick) {
                Icon(
                    imageVector = Icons.Default.Sort,
                    contentDescription = "排序"
                )
            }

            TodoListOverflowMenu()
        },
        modifier = modifier
    )
}

@Composable
private fun SearchTextField(
    query: String,
    onQueryChange: (String) -> Unit,
    onSearchActiveChange: (Boolean) -> Unit,
    placeholder: String,
    modifier: Modifier = Modifier
) {
    OutlinedTextField(
        value = query,
        onValueChange = onQueryChange,
        placeholder = { Text(placeholder) },
        leadingIcon = {
            Icon(
                imageVector = Icons.Default.Search,
                contentDescription = null
            )
        },
        trailingIcon = {
            if (query.isNotEmpty()) {
                IconButton(onClick = { onQueryChange("") }) {
                    Icon(
                        imageVector = Icons.Default.Clear,
                        contentDescription = "清除搜索"
                    )
                }
            } else {
                IconButton(onClick = { onSearchActiveChange(false) }) {
                    Icon(
                        imageVector = Icons.Default.Close,
                        contentDescription = "关闭搜索"
                    )
                }
            }
        },
        singleLine = true,
        modifier = modifier.fillMaxWidth()
    )
}

@Composable
fun TodoFilterChips(
    currentFilter: TodoFilter,
    onFilterChange: (TodoFilter) -> Unit,
    modifier: Modifier = Modifier
) {
    LazyRow(
        horizontalArrangement = Arrangement.spacedBy(TodoDimensions.SpacingSmall),
        contentPadding = PaddingValues(horizontal = TodoDimensions.SpacingMedium),
        modifier = modifier
    ) {
        items(TodoFilter.values()) { filter ->
            FilterChip(
                selected = currentFilter == filter,
                onClick = { onFilterChange(filter) },
                label = {
                    Text(
                        text = filter.displayName,
                        style = MaterialTheme.typography.labelMedium
                    )
                },
                leadingIcon = if (currentFilter == filter) {
                    {
                        Icon(
                            imageVector = Icons.Default.Check,
                            contentDescription = null,
                            modifier = Modifier.size(TodoDimensions.IconSizeSmall)
                        )
                    }
                } else null
            )
        }
    }
}

// ui/components/TodoListContent.kt
@Composable
fun TodoListContent(
    uiState: TodoListUiState,
    onAction: (TodoListAction) -> Unit,
    modifier: Modifier = Modifier
) {
    Box(modifier = modifier.fillMaxSize()) {
        when {
            uiState.isLoading -> {
                LoadingIndicator(
                    modifier = Modifier.align(Alignment.Center)
                )
            }

            uiState.error != null -> {
                ErrorMessage(
                    error = uiState.error,
                    onRetry = { onAction(TodoListAction.RefreshTodoItems) },
                    modifier = Modifier.align(Alignment.Center)
                )
            }

            uiState.todoItems.isEmpty() -> {
                EmptyStateMessage(
                    currentFilter = uiState.currentFilter,
                    onCreateTodo = { onAction(TodoListAction.NavigateToCreate) },
                    modifier = Modifier.align(Alignment.Center)
                )
            }

            else -> {
                TodoItemsList(
                    todoItems = uiState.todoItems,
                    selectedItems = uiState.selectedTodoIds,
                    isSelectionMode = uiState.isSelectionMode,
                    onAction = onAction
                )
            }
        }

        // 选择模式底部操作栏
        if (uiState.isSelectionMode) {
            SelectionBottomBar(
                selectedCount = uiState.selectedCount,
                onAction = onAction,
                modifier = Modifier.align(Alignment.BottomCenter)
            )
        }
    }
}

@Composable
private fun TodoItemsList(
    todoItems: List<TodoItem>,
    selectedItems: Set<String>,
    isSelectionMode: Boolean,
    onAction: (TodoListAction) -> Unit,
    modifier: Modifier = Modifier
) {
    LazyColumn(
        verticalArrangement = Arrangement.spacedBy(TodoDimensions.SpacingSmall),
        contentPadding = PaddingValues(
            horizontal = TodoDimensions.SpacingMedium,
            vertical = TodoDimensions.SpacingSmall
        ),
        modifier = modifier
    ) {
        items(
            items = todoItems,
            key = { it.id }
        ) { todoItem ->
            TodoItemCard(
                todoItem = todoItem,
                isSelected = selectedItems.contains(todoItem.id),
                isSelectionMode = isSelectionMode,
                onItemClick = {
                    if (isSelectionMode) {
                        if (selectedItems.contains(todoItem.id)) {
                            onAction(TodoListAction.DeselectTodoItem(todoItem.id))
                        } else {
                            onAction(TodoListAction.SelectTodoItem(todoItem.id))
                        }
                    } else {
                        onAction(TodoListAction.NavigateToDetail(todoItem.id))
                    }
                },
                onItemLongClick = {
                    if (!isSelectionMode) {
                        onAction(TodoListAction.EnterSelectionMode(todoItem.id))
                    }
                },
                onCompleteClick = {
                    onAction(TodoListAction.CompleteTodoItem(todoItem.id))
                },
                onSnoozeClick = { minutes ->
                    onAction(TodoListAction.SnoozeTodoItem(todoItem.id, minutes))
                },
                onDeleteClick = {
                    onAction(TodoListAction.DeleteTodoItem(todoItem.id))
                },
                modifier = Modifier.animateItemPlacement()
            )
        }

        // 底部间距，避免被FAB遮挡
        item {
            Spacer(modifier = Modifier.height(80.dp))
        }
    }
}

// ui/components/TodoItemCard.kt
@Composable
fun TodoItemCard(
    todoItem: TodoItem,
    isSelected: Boolean = false,
    isSelectionMode: Boolean = false,
    onItemClick: () -> Unit,
    onItemLongClick: () -> Unit = {},
    onCompleteClick: () -> Unit,
    onSnoozeClick: (Int) -> Unit = {},
    onDeleteClick: () -> Unit = {},
    modifier: Modifier = Modifier
) {
    val interactionSource = remember { MutableInteractionSource() }
    var showActionMenu by remember { mutableStateOf(false) }

    Card(
        modifier = modifier
            .fillMaxWidth()
            .combinedClickable(
                interactionSource = interactionSource,
                indication = ripple(),
                onClick = onItemClick,
                onLongClick = onItemLongClick
            )
            .then(
                if (isSelected) {
                    Modifier.border(
                        width = 2.dp,
                        color = MaterialTheme.colorScheme.primary,
                        shape = MaterialTheme.shapes.medium
                    )
                } else Modifier
            ),
        colors = CardDefaults.cardColors(
            containerColor = if (isSelected) {
                MaterialTheme.colorScheme.primaryContainer.copy(alpha = 0.3f)
            } else {
                MaterialTheme.colorScheme.surface
            }
        ),
        elevation = CardDefaults.cardElevation(
            defaultElevation = if (isSelected) 8.dp else 4.dp
        )
    ) {
        Column(
            modifier = Modifier.padding(TodoDimensions.SpacingMedium)
        ) {
            // 顶部行：选择指示器/完成按钮 + 标题 + 优先级标签
            Row(
                verticalAlignment = Alignment.CenterVertically,
                modifier = Modifier.fillMaxWidth()
            ) {
                // 左侧：选择指示器或完成按钮
                if (isSelectionMode) {
                    Checkbox(
                        checked = isSelected,
                        onCheckedChange = { onItemClick() }
                    )
                } else {
                    TodoCompleteButton(
                        isCompleted = todoItem.isCompleted(),
                        onClick = onCompleteClick
                    )
                }

                Spacer(modifier = Modifier.width(TodoDimensions.SpacingMedium))

                // 中间：标题
                Text(
                    text = todoItem.title,
                    style = MaterialTheme.typography.bodyLarge.copy(
                        textDecoration = if (todoItem.isCompleted()) {
                            TextDecoration.LineThrough
                        } else null
                    ),
                    color = if (todoItem.isCompleted()) {
                        MaterialTheme.colorScheme.onSurface.copy(alpha = 0.6f)
                    } else {
                        MaterialTheme.colorScheme.onSurface
                    },
                    maxLines = 2,
                    overflow = TextOverflow.Ellipsis,
                    modifier = Modifier.weight(1f)
                )

                Spacer(modifier = Modifier.width(TodoDimensions.SpacingSmall))

                // 右侧：优先级标签
                TodoPriorityChip(
                    priority = todoItem.getPriorityEnum(),
                    size = TodoChipSize.Small
                )

                // 操作菜单按钮
                if (!isSelectionMode) {
                    IconButton(
                        onClick = { showActionMenu = true }
                    ) {
                        Icon(
                            imageVector = Icons.Default.MoreVert,
                            contentDescription = "更多操作"
                        )
                    }
                }
            }

            // 内容描述（如果有）
            if (todoItem.content.isNotBlank()) {
                Spacer(modifier = Modifier.height(TodoDimensions.SpacingSmall))
                Text(
                    text = todoItem.content,
                    style = MaterialTheme.typography.bodyMedium,
                    color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.7f),
                    maxLines = 3,
                    overflow = TextOverflow.Ellipsis
                )
            }

            Spacer(modifier = Modifier.height(TodoDimensions.SpacingMedium))

            // 底部信息行：分类、时间、状态
            Row(
                verticalAlignment = Alignment.CenterVertically,
                horizontalArrangement = Arrangement.spacedBy(TodoDimensions.SpacingSmall),
                modifier = Modifier.fillMaxWidth()
            ) {
                // 分类标签
                TodoCategoryChip(
                    category = todoItem.getCategoryEnum(),
                    size = TodoChipSize.Small
                )

                // 时间信息
                TodoTimeChip(
                    timestamp = todoItem.triggerTime,
                    isOverdue = todoItem.isOverdue(),
                    size = TodoChipSize.Small
                )

                Spacer(modifier = Modifier.weight(1f))

                // 状态指示器
                TodoStatusIndicator(
                    status = TodoStatus.valueOf(todoItem.status),
                    size = TodoIndicatorSize.Small
                )
            }

            // 标签行（如果有标签）
            val tags = todoItem.getTagList()
            if (tags.isNotEmpty()) {
                Spacer(modifier = Modifier.height(TodoDimensions.SpacingSmall))
                LazyRow(
                    horizontalArrangement = Arrangement.spacedBy(TodoDimensions.SpacingSmall)
                ) {
                    items(tags.take(3)) { tag ->
                        TodoTagChip(
                            tag = tag,
                            size = TodoChipSize.Small
                        )
                    }
                    if (tags.size > 3) {
                        item {
                            Text(
                                text = "+${tags.size - 3}",
                                style = MaterialTheme.typography.labelSmall,
                                color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.6f)
                            )
                        }
                    }
                }
            }
        }
    }

    // 操作菜单
    TodoItemActionMenu(
        expanded = showActionMenu,
        onDismiss = { showActionMenu = false },
        onSnooze = onSnoozeClick,
        onDelete = onDeleteClick,
        todoItem = todoItem
    )
}

@Composable

        Text(
            text = getEmptyStateTitle(currentFilter),

            color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.6f),
            textAlign = TextAlign.Center
        )

        Spacer(modifier = Modifier.height(TodoDimensions.SpacingSmall))

        // 空状态描述
        Text(
            text = getEmptyStateDescription(currentFilter),
            style = MaterialTheme.typography.bodyMedium,
            color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.4f),
            textAlign = TextAlign.Center
        )

        // 创建按钮（仅在全部或待完成筛选时显示）
        if (currentFilter in listOf(TodoFilter.ALL, TodoFilter.PENDING)) {
            Spacer(modifier = Modifier.height(TodoDimensions.SpacingLarge))

            Button(
                onClick = onCreateTodo,
                modifier = Modifier.fillMaxWidth(0.6f)

                Icon(
                    imageVector = Icons.Default.Add,
                    contentDescription = null,
                    modifier = Modifier.size(18.dp)
                )
                Spacer(modifier = Modifier.width(TodoDimensions.SpacingSmall))
                Text("创建第一个待办")
            }
        }
    }
}

        TodoFilter.ALL -> "还没有待办事项"

        TodoFilter.TODAY -> "今天没有待办事项"
        TodoFilter.THIS_WEEK -> "本周没有待办事项"

@Composable
fun TodoItemSwipeActions(
    todoItem: TodoItem,
    onComplete: () -> Unit,
    onDelete: () -> Unit,
    content: @Composable () -> Unit
) {
    val dismissState = rememberDismissState(
        confirmValueChange = { dismissValue ->
            when (dismissValue) {
                DismissValue.DismissedToEnd -> {
                    onComplete()
                    true
                }
                DismissValue.DismissedToStart -> {
                    onDelete()
                    true
                }
                else -> false
            }
        }
    )

    SwipeToDismiss(
        state = dismissState,
        background = {
            SwipeBackground(
                dismissState = dismissState,
                todoItem = todoItem
            )
        },
        dismissContent = { content() }
    )
}

@Composable
private fun SwipeBackground(
    dismissState: DismissState,
    todoItem: TodoItem
) {
    val direction = dismissState.dismissDirection ?: return
    val color by animateColorAsState(
        targetValue = when (direction) {
            DismissDirection.StartToEnd -> TodoColors.Success
            DismissDirection.EndToStart -> TodoColors.Error
        },
        animationSpec = tween(300)
    )

    Box(
        modifier = Modifier
            .fillMaxSize()
            .background(color)
            .padding(horizontal = 20.dp),

@Composable
fun OptimizedTodoList(
    todoItems: List<TodoItemUiState>,
    onAction: (TodoListAction) -> Unit,
    modifier: Modifier = Modifier
) {
    // 使用 derivedStateOf 避免不必要的重组
    val groupedItems by remember {
        derivedStateOf {
            todoItems.groupBy { item ->
                when {
                    item.isToday() -> "今天"
                    item.isOverdue() -> "过期"
                    item.isThisWeek() -> "本周"
                    else -> "其他"
                }
            }
        }
    }

    LazyColumn(
        modifier = modifier,
        verticalArrangement = Arrangement.spacedBy(4.dp)
    ) {
        groupedItems.forEach { (group, items) ->
            stickyHeader {
                TodoGroupHeader(
                    title = group,
                    count = items.size
                )
            }

            items(
                items = items,
                key = { it.id }
            ) { item ->
                TodoItemCard(
                    todoItem = item,
                    onItemClick = remember { { onAction(TodoListAction.SelectItem(item.id)) } },
                    modifier = Modifier.animateItemPlacement()
                )
            }
        }
    }
}

@Composable
fun AccessibleTodoItemCard(
    todoItem: TodoItem,
    onItemClick: () -> Unit,
    onCompleteClick: () -> Unit,
    modifier: Modifier = Modifier
) {
    Card(
        modifier = modifier
            .fillMaxWidth()
            .clickable { onItemClick() }
            .semantics {
                contentDescription = buildString {
                    append("待办事项: ${todoItem.title}")
                    if (todoItem.content.isNotBlank()) {
                        append(", 内容: ${todoItem.content}")
                    }
                    append(", 状态: ${if (todoItem.isCompleted()) "已完成" else "待完成"}")
                    append(", 优先级: ${todoItem.priority}")
                    append(", 时间: ${formatAccessibleTime(todoItem.triggerTime)}")
                    if (todoItem.isOverdue()) {
                        append(", 已过期")
                    }
                }

                customActions = listOf(
                    CustomAccessibilityAction(
                        label = if (todoItem.isCompleted()) "取消完成" else "标记完成",
                        action = { onCompleteClick(); true }
                    )
                )
            }
    ) {
        // 卡片内容
    }
}

private fun formatAccessibleTime(timestamp: Long): String {
    val now = System.currentTimeMillis()
    val diff = timestamp - now

    return when {
        diff < 0 -> "已过期"
        diff < 60 * 60 * 1000 -> "1小时内"
        diff < 24 * 60 * 60 * 1000 -> "今天"
        diff < 7 * 24 * 60 * 60 * 1000 -> "本周"
        else -> SimpleDateFormat("MM月dd日", Locale.getDefault()).format(Date(timestamp))
    }
}
